Quick Start

Datadriver works in conjunction with the WebdriverIO API and the grunt-webdriver plugin. We'll take you through setting up a project step-by-step, or you can just download the code.

  1. In a new directory, create a package.json file with the following contents. 
    {
  "version": "1.0.0",
  "description": "Get up and running with datadriver.io",
  "engines": {
    "node": ">= 0.12.0"
  },
  "devDependencies": {
    "datadriver": "0.1.4",
    "grunt": "0.4.5",
    "grunt-webdriver": "0.5.2"
  }
}
  2. Create a gruntfile.js file with the following contents. 
    var gruntConfig = {
  webdriver: {
    options: {
      desiredCapabilities: {
        browserName: 'chrome',
        name: 'Documentation Tests',
        viewports: {
          small: {
            width: 430,
            height: 1000
          },
          medium: {
            width: 700,
            height: 1000
          },
          large: {
            width: 1280,
            height: 1000
          }
        }
      }
    },
    home: {
      tests: ['tests/*.js']
    }
  }
};

    
module.exports = function(grunt) {
  grunt.initConfig(gruntConfig);
  grunt.loadNpmTasks('grunt-webdriver');
};
  3. In your root directory, create another directory called tests.
  4. Inside of your tests directory, create a file called duck-duck-go.js with the following contents. 
    'use strict';

    
var drive = require('datadriver');

    
drive.go({

    
  "config": {
    "description": "Duck Duck Go Test",
    "suites": [
      {
        "description": "Search for DataDriver.io",
        "tests": ["duck-duck-go"],
        "viewports": { small: true }
      }
    ]
  },

    
  "duck-duck-go": {
    "steps": [
      {
        "selector": "#search_form_input_homepage",
        "execute": [{ "action": "type", "text": "Datadriver.io" }]
      },
      {
        "selector": "#search_button_homepage",
        "execute": [{ "action": "click" }],
      },
      {
        "verify": [
          {
            "method": "verifyPageTitle",
            "asserts": { "title": "Datadriver.io at DuckDuckGo" }
          }
        ]
      }
    ]
  }

    
});
  5. From your command line program, navigate to your root directory and run the test. 
    cd <your root directory path>
npm install
grunt webdriver --url=https://duckduckgo.com
  6. Your output should look something like this: 
    $ grunt webdriver:home --url=http://duckduckgo.com
Running "webdriver:home" (webdriver) task
Duck Duck Go Test
  Rendered in a small viewport
    Search for DataDriver.io
      ⚙ executing "setValue" on "#search_form_input_homepage".
      ⚙ clicking the "#search_button_homepage" element.
      ✓ the "title" of this page should equal "Datadriver.io at DuckDuckGo"
